import streamlit as st import math # --- CONFIGURACIÓN DE LA PÁGINA --- st.set_page_config(page_title="Calculadora Pothenot", layout="centered") # --- ESTILOS VISUALES --- st.title("Calculadora Topográfica: Método Pothenot") st.markdown("Calcula las coordenadas de tu estación (P) a partir de tres puntos conocidos (A, B, C) y los ángulos observados.") # --- FUNCIONES MATEMÁTICAS --- def dms_a_decimal(grados, minutos, segundos): return grados + (minutos / 60.0) + (segundos / 3600.0) def decimal_a_dms(decimal): g = int(decimal) m = int((decimal - g) * 60) s = (decimal - g - m / 60.0) * 3600 return f"{g}° {m}' {s:.2f}\"" def calcular_azimut(x1, y1, x2, y2): azimut_rad = math.atan2(x2 - x1, y2 - y1) azimut_deg = math.degrees(azimut_rad) return (azimut_deg + 360) % 360 # --- INTERFAZ DE USUARIO (Inputs) --- st.header("1. Coordenadas Conocidas (UTM)") col1, col2 = st.columns(2) with col1: st.subheader("Punto A") XA = st.number_input("Este (X) - A", value=274600.21, format="%.3f") YA = st.number_input("Norte (Y) - A", value=8662610.80, format="%.3f") st.subheader("Punto C") XC = st.number_input("Este (X) - C", value=275330.60, format="%.3f") YC = st.number_input("Norte (Y) - C", value=8662624.10, format="%.3f") with col2: st.subheader("Punto B (Central)") XB = st.number_input("Este (X) - B", value=274881.40, format="%.3f") YB = st.number_input("Norte (Y) - B", value=8662582.60, format="%.3f") st.header("2. Ángulos Observados") col3, col4 = st.columns(2) with col3: st.subheader("Ángulo Alfa (α)") a_g = st.number_input("Grados α", value=30, step=1) a_m = st.number_input("Minutos α", value=12, step=1, max_value=59) a_s = st.number_input("Segundos α", value=10.0, step=0.1, max_value=59.99) with col4: st.subheader("Ángulo Beta (β)") b_g = st.number_input("Grados β", value=49, step=1) b_m = st.number_input("Minutos β", value=53, step=1, max_value=59) b_s = st.number_input("Segundos β", value=40.0, step=0.1, max_value=59.99) # --- BOTÓN DE CÁLCULO --- if st.button("Calcular Coordenadas de P", type="primary"): # --- PROCESAMIENTO MATEMÁTICO --- try: alpha_deg = dms_a_decimal(a_g, a_m, a_s) beta_deg = dms_a_decimal(b_g, b_m, b_s) alpha = math.radians(alpha_deg) beta = math.radians(beta_deg) AB = math.hypot(XB - XA, YB - YA) BC = math.hypot(XC - XB, YC - YB) Z_AB = calcular_azimut(XA, YA, XB, YB) Z_CB = calcular_azimut(XC, YC, XB, YB) Z_BA = (Z_AB + 180) % 360 Z_BC = (Z_CB + 180) % 360 theta_deg = (Z_BA - Z_BC) % 360 theta = math.radians(theta_deg) suma_AC_deg = 360 - (theta_deg + alpha_deg + beta_deg) suma_AC = math.radians(suma_AC_deg) cot_suma = 1 / math.tan(suma_AC) ctg_A = (AB * math.sin(beta)) / (BC * math.sin(alpha) * math.sin(suma_AC)) + cot_suma A_rad = math.atan(1 / ctg_A) A_deg = math.degrees(A_rad) if A_deg < 0: A_deg += 180 C_deg = suma_AC_deg - A_deg theta1_deg = 180 - (A_deg + alpha_deg) theta2_deg = 180 - (C_deg + beta_deg) theta1 = math.radians(theta1_deg) theta2 = math.radians(theta2_deg) AP = AB * math.sin(theta1) / math.sin(alpha) CP = BC * math.sin(theta2) / math.sin(beta) Z_AP = (Z_AB + A_deg) % 360 Z_CP = (Z_CB - C_deg) % 360 XP_A = XA + AP * math.sin(math.radians(Z_AP)) YP_A = YA + AP * math.cos(math.radians(Z_AP)) XP_C = XC + CP * math.sin(math.radians(Z_CP)) YP_C = YC + CP * math.cos(math.radians(Z_CP)) error_cierre = math.hypot(XP_A - XP_C, YP_A - YP_C) # --- MOSTRAR RESULTADOS --- st.divider() st.header("Resultados del Levantamiento") if error_cierre < 0.005: st.success("Cierre Matemático Perfecto (Error < 5mm)") # Tarjeta de coordenadas destacada st.info(f"**COORDENADAS DE LA ESTACIÓN (P):**\n\n**ESTE (X):** `{XP_A:.3f}`\n\n**NORTE (Y):** `{YP_A:.3f}`") # Datos adicionales desplegables with st.expander("Ver desglose analítico"): st.write(f"- **Ángulo calculado en A:** {decimal_a_dms(A_deg)}") st.write(f"- **Ángulo calculado en C:** {decimal_a_dms(C_deg)}") st.write(f"- **Distancia a P desde A:** {AP:.3f} m") st.write(f"- **Distancia a P desde C:** {CP:.3f} m") else: st.error(f"Atención: Hay una discrepancia de {error_cierre:.4f}m en la comprobación matemática.
